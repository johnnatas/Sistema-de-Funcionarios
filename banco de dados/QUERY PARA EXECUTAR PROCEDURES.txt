###EXECUTA PROCEDURE QUE SELECIONA A LINHA DO ID DIGITADO###
set @V_SALARIO = 0;

set @V_NOME = '';

set @V_RESPOSTA = '';

call `database`.SELECTTAB(1, @V_RESPOSTA, @V_NOME, @V_SALARIO);

select @V_NOME, @V_SALARIO, @V_RESPOSTA;

set @V_RESPOSTA = '';

select @V_RESPOSTA;



###EXECUTA PROCEDURE QUE DELETA UMA LINHA DA TABELA FUNCIONARIO###
